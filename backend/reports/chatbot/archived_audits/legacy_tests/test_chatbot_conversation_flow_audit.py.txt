def _normalize_route(value: str) -> str:
    return str(value or "").strip().upper()


def _source_types(payload: dict[str, Any]) -> set[str]:
    result: set[str] = set()
    for src in payload.get("sources") or []:
        token = str((src or {}).get("type") or "").strip().lower()
        if token:
            result.add(token)
    return result


def _block_types(payload: dict[str, Any]) -> set[str]:
    result: set[str] = set()
    for block in payload.get("response_blocks") or []:
        token = str((block or {}).get("type") or "").strip().lower()
        if token:
            result.add(token)
    return result


def _extract_entities(payload: dict[str, Any]) -> dict[str, Any]:
    metadata = payload.get("metadata") if isinstance(payload, dict) else None
    if not isinstance(metadata, dict):
        return {}
    entities = metadata.get("detected_entities")
    return entities if isinstance(entities, dict) else {}


def _numbers_from_text(blob: str) -> list[float]:
    values: list[float] = []
    for token in re.findall(r"\d+(?:[\.,]\d+)?", blob):
        try:
            values.append(float(token.replace(",", ".")))
        except ValueError:
            continue
    return values


def _contains_close_number(blob: str, expected: float, *, tolerance_ratio: float = 0.1, tolerance_abs: float = 1.0) -> bool:
    threshold = max(tolerance_abs, abs(expected) * tolerance_ratio)
    for value in _numbers_from_text(blob):
        if abs(value - expected) <= threshold:
            return True
    return False


def _is_french_like(answer: str) -> bool:
    text_l = " ".join(str(answer or "").lower().split())
    if not text_l:
        return False
    fr_tokens = {"le", "la", "les", "des", "pour", "avec", "et", "est", "sont", "données", "coopérative", "lot", "perte"}
    en_tokens = {"the", "and", "with", "for", "data", "batch", "loss", "stock", "please", "should"}
    fr_hits = sum(1 for tok in fr_tokens if tok in text_l)
    en_hits = sum(1 for tok in en_tokens if tok in text_l)
    return fr_hits >= 1 or en_hits <= 2


def _table_exists(db: Session, table_name: str) -> bool:
    try:
        db.execute(text(f"SELECT 1 FROM {table_name} LIMIT 1"))
        return True
    except Exception:
        return False


def _pick_first(values: list[str], fallback: str, index: int = 0) -> str:
    if not values:
        return fallback
    return values[index % len(values)]


def _canonical_product(name: str) -> str:
    lowered = str(name or "").lower().strip()
    if lowered in {"mangue", "mango"}:
        return "mango"
    if lowered in {"arachide", "peanut"}:
        return "peanut"
    if lowered in {"mil", "millet"}:
        return "millet"
    return lowered


def _build_truth_snapshot(db: Session, cooperative_id: Any) -> dict[str, Any]:
    products = [
        str(row[0])
        for row in db.execute(
            text("SELECT name FROM products WHERE cooperative_id = :coop_id ORDER BY name ASC LIMIT 10"),
            {"coop_id": cooperative_id},
        ).fetchall()
        if str(row[0] or "").strip()
    ]
    batches = [
        str(row[0])
        for row in db.execute(
            text("SELECT code FROM batches WHERE cooperative_id = :coop_id ORDER BY creation_date DESC LIMIT 20"),
            {"coop_id": cooperative_id},
        ).fetchall()
        if str(row[0] or "").strip()
    ]

    members_count = int(
        db.execute(text("SELECT COUNT(*) FROM members WHERE cooperative_id = :coop_id"), {"coop_id": cooperative_id}).scalar() or 0
    )
    parcel_count = int(
        db.execute(text("SELECT COUNT(*) FROM parcels WHERE cooperative_id = :coop_id"), {"coop_id": cooperative_id}).scalar() or 0
    )
    input_total_kg = float(
        db.execute(text("SELECT COALESCE(SUM(quantity), 0) FROM inputs WHERE cooperative_id = :coop_id"), {"coop_id": cooperative_id}).scalar()
        or 0.0
    )
    stock_total_kg = float(
        db.execute(
            text(
                """
                SELECT COALESCE(SUM(COALESCE(total_stock_kg, 0) - COALESCE(reserved_in_lots_kg, 0)), 0)
                FROM stocks
                WHERE cooperative_id = :coop_id
                """
            ),
            {"coop_id": cooperative_id},
        ).scalar()
        or 0.0
    )
    lots_open_count = int(
        db.execute(
            text(
                """
                SELECT COUNT(*)
                FROM batches
                WHERE cooperative_id = :coop_id
                  AND status NOT IN ('completed', 'cancelled', 'COMPLETED', 'CANCELLED')
                """
            ),
            {"coop_id": cooperative_id},
        ).scalar()
        or 0
    )
    low_eff_count = int(
        db.execute(
            text(
                """
                SELECT COUNT(*)
                FROM batches
                WHERE cooperative_id = :coop_id
                  AND initial_qty > 0
                  AND (current_qty / initial_qty) < 0.85
                """
            ),
            {"coop_id": cooperative_id},
        ).scalar()
        or 0
    )

    top_members = [
        dict(row._mapping)
        for row in db.execute(
            text(
                """
                SELECT m.full_name, m.code AS member_code, COALESCE(SUM(i.quantity), 0.0) AS total_kg
                FROM members m
                JOIN inputs i ON i.member_id = m.id
                WHERE m.cooperative_id = :coop_id
                  AND i.cooperative_id = :coop_id
                GROUP BY m.id, m.full_name, m.code
                ORDER BY total_kg DESC
                LIMIT 10
                """
            ),
            {"coop_id": cooperative_id},
        ).fetchall()
    ]

    top_loss_lot = db.execute(
        text(
            """
            SELECT code,
                   CASE WHEN initial_qty > 0 THEN ((initial_qty - current_qty) / initial_qty) * 100 ELSE 0 END AS loss_pct
            FROM batches
            WHERE cooperative_id = :coop_id
            ORDER BY loss_pct DESC, code ASC
            LIMIT 1
            """
        ),
        {"coop_id": cooperative_id},
    ).fetchone()

    material = dict(
        db.execute(
            text(
                """
                SELECT COALESCE(SUM(initial_qty), 0) AS in_kg,
                       COALESCE(SUM(current_qty), 0) AS out_kg
                FROM batches
                WHERE cooperative_id = :coop_id
                """
            ),
            {"coop_id": cooperative_id},
        ).one()._mapping
    )
    in_kg = float(material.get("in_kg") or 0.0)
    out_kg = float(material.get("out_kg") or 0.0)
    loss_pct = ((in_kg - out_kg) / in_kg * 100.0) if in_kg > 0 else 0.0

    stage_losses = [
        dict(row._mapping)
        for row in db.execute(
            text(
                """
                SELECT ps.type AS stage, COALESCE(SUM(ps.loss_value), 0.0) AS total_loss_kg
                FROM process_steps ps
                JOIN batches b ON b.id = ps.batch_id
                WHERE b.cooperative_id = :coop_id
                GROUP BY ps.type
                ORDER BY total_loss_kg DESC
                LIMIT 6
                """
            ),
            {"coop_id": cooperative_id},
        ).fetchall()
    ]

    sql_risk_count = int(
        db.execute(
            text(
                """
                SELECT COUNT(*)
                FROM batches
                WHERE cooperative_id = :coop_id
                  AND initial_qty > 0
                  AND ((initial_qty - current_qty) / initial_qty) * 100 >= :threshold
                """
            ),
            {"coop_id": cooperative_id, "threshold": float(settings.anomaly_loss_threshold)},
        ).scalar()
        or 0
    )

    rag_count = 0
    rag_keywords = 0
    if _table_exists(db, "rag_chunks"):
        rag_count = int(
            db.execute(
                text("SELECT COUNT(*) FROM rag_chunks WHERE cooperative_id = :coop_id"),
                {"coop_id": cooperative_id},
            ).scalar()
            or 0
        )
        rag_keywords = int(
            db.execute(
                text(
                    """
                    SELECT COUNT(*)
                    FROM rag_chunks
                    WHERE cooperative_id = :coop_id
                      AND lower(content) ~ '(séchage|sechage|tri|stockage|humidit|perte)'
                    """
                ),
                {"coop_id": cooperative_id},
            ).scalar()
            or 0
        )

    reco_count = 0
    if _table_exists(db, "recommendations"):
        reco_count = int(
            db.execute(
                text(
                    """
                    SELECT COUNT(*)
                    FROM recommendations r
                    JOIN batches b ON b.id = r.batch_id
                    WHERE b.cooperative_id = :coop_id
                    """
                ),
                {"coop_id": cooperative_id},
            ).scalar()
            or 0
        )

    business_availability = {
        "commercial_orders": _table_exists(db, "commercial_orders"),
        "commercial_invoices": _table_exists(db, "commercial_invoices"),
        "treasury_transactions": _table_exists(db, "treasury_transactions"),
        "global_charges": _table_exists(db, "global_charges"),
    }

    commercial_orders_count = 0
    invoices_count = 0
    treasury_count = 0
    charges_count = 0
    if business_availability["commercial_orders"]:
        commercial_orders_count = int(
            db.execute(
                text("SELECT COUNT(*) FROM commercial_orders WHERE cooperative_id = :coop_id"),
                {"coop_id": cooperative_id},
            ).scalar()
            or 0
        )
    if business_availability["commercial_invoices"]:
        invoices_count = int(
            db.execute(
                text("SELECT COUNT(*) FROM commercial_invoices WHERE cooperative_id = :coop_id"),
                {"coop_id": cooperative_id},
            ).scalar()
            or 0
        )
    if business_availability["treasury_transactions"]:
        treasury_count = int(
            db.execute(
                text("SELECT COUNT(*) FROM treasury_transactions WHERE cooperative_id = :coop_id"),
                {"coop_id": cooperative_id},
            ).scalar()
            or 0
        )
    if business_availability["global_charges"]:
        charges_count = int(
            db.execute(
                text("SELECT COUNT(*) FROM global_charges WHERE cooperative_id = :coop_id"),
                {"coop_id": cooperative_id},

    return {
        "products": products,
        "lots": batches,
        "members_count": members_count,
        "parcel_count": parcel_count,
        "input_total_kg": input_total_kg,
        "stock_total_kg": stock_total_kg,
        "lots_open_count": lots_open_count,
        "low_eff_count": low_eff_count,
        "top_members": top_members,
        "top_loss_lot": dict(top_loss_lot._mapping) if top_loss_lot else None,
        "material": {"in_kg": in_kg, "out_kg": out_kg, "loss_pct": loss_pct},
        "stage_losses": stage_losses,
        "sql_risk_count": sql_risk_count,
        "rag_count": rag_count,
        "rag_keywords": rag_keywords,
        "recommendations_count": reco_count,
        "business_availability": business_availability,
        "commercial_orders_count": commercial_orders_count,

        "treasury_count": treasury_count,

    }

def _expected_product_in_entities(entities: dict[str, Any], expected_product: str) -> bool:
    products = entities.get("product") or []
    tokens = {_canonical_product(str(item)) for item in products if str(item).strip()}
    return expected_product in tokens


def _fact_check(case: ConversationCase, truth: dict[str, Any], payload: dict[str, Any]) -> tuple[bool, str]:
    answer = str(payload.get("answer") or "")
    blob = f"{answer}\n{json.dumps(payload.get('response_blocks') or [], ensure_ascii=False)}"
    lower_blob = blob.lower()

    if case.fact_check is None:
        return True, "n/a"

    if case.fact_check == "members_count":
        expected = float(truth.get("members_count") or 0)
        return _contains_close_number(blob, expected), f"expected members_count={expected:.0f}"

    if case.fact_check == "member_ranking":
        top = (truth.get("top_members") or [None])[0]
        if not top:
            return False, "no top member truth"
        name_ok = str(top.get("full_name") or "").lower() in lower_blob
        kg_ok = _contains_close_number(blob, float(top.get("total_kg") or 0.0), tolerance_ratio=0.12, tolerance_abs=2.0)
        return (name_ok and kg_ok), f"expected top_member={top.get('full_name')} ({float(top.get('total_kg') or 0.0):.1f} kg)"

    if case.fact_check == "parcel_count":
        expected = float(truth.get("parcel_count") or 0)
        return _contains_close_number(blob, expected), f"expected parcel_count={expected:.0f}"

    if case.fact_check == "inputs_total":
        expected = float(truth.get("input_total_kg") or 0.0)
        return _contains_close_number(blob, expected, tolerance_ratio=0.15, tolerance_abs=5.0), f"expected inputs_total={expected:.1f} kg"

    if case.fact_check == "stock_total":
        expected = float(truth.get("stock_total_kg") or 0.0)
        return _contains_close_number(blob, expected, tolerance_ratio=0.2, tolerance_abs=5.0), f"expected stock_total={expected:.1f} kg"

    if case.fact_check == "batch_open_count":
        expected = float(truth.get("lots_open_count") or 0)
        return _contains_close_number(blob, expected), f"expected lots_open_count={expected:.0f}"

    if case.fact_check == "top_stage":
        stage_rows = truth.get("stage_losses") or []
        if not stage_rows:
            return False, "no process stage truth"
        stage = str(stage_rows[0].get("stage") or "").lower()
        return stage in lower_blob and ("perte" in lower_blob or "loss" in lower_blob), f"expected top_stage={stage}"

    if case.fact_check == "material_global":
        material = truth.get("material") or {}
        loss_pct = float(material.get("loss_pct") or 0.0)
        in_kg = float(material.get("in_kg") or 0.0)
        out_kg = float(material.get("out_kg") or 0.0)
        ok = (
            _contains_close_number(blob, loss_pct, tolerance_ratio=0.2, tolerance_abs=1.5)
            or _contains_close_number(blob, in_kg, tolerance_ratio=0.2, tolerance_abs=5.0)
            or _contains_close_number(blob, out_kg, tolerance_ratio=0.2, tolerance_abs=5.0)
        )
        return ok, f"expected material in={in_kg:.1f} out={out_kg:.1f} loss={loss_pct:.1f}%"

    if case.fact_check == "low_eff_count":
        expected = float(truth.get("low_eff_count") or 0)
        ok = _contains_close_number(blob, expected) and any(token in lower_blob for token in ("effic", "faible", "perte", "rendement"))
        return ok, f"expected low_eff_count={expected:.0f}"

    if case.fact_check == "top_loss_lot":
        top = truth.get("top_loss_lot") or {}
        code = str(top.get("code") or "").lower()
        if not code:
            return False, "no top_loss_lot truth"
        return code in lower_blob and ("perte" in lower_blob or "loss" in lower_blob), f"expected top_loss_lot={code}"

    if case.fact_check == "ml_risk_semantic":
        semantic_ok = any(token in lower_blob for token in ("risque", "anomal", "ml", "signal"))
        return semantic_ok, f"expected risk semantics; sql_risk_count={truth.get('sql_risk_count')}"

    if case.fact_check == "rag_presence":
        semantic_ok = any(token in lower_blob for token in ("séch", "sech", "tri", "pratique", "référence", "humid", "stockage"))
        return semantic_ok, f"expected rag semantics; rag_keywords={truth.get('rag_keywords')}"

    if case.fact_check == "recommendation_semantic":
        semantic_ok = any(token in lower_blob for token in ("recommand", "action", "priorit", "amélior", "amelior", "rédu"))
        return semantic_ok, f"expected recommendation semantics; recommendation_rows={truth.get('recommendations_count')}"

    if case.fact_check == "business_semantic":
        semantic_ok = any(token in lower_blob for token in ("commande", "facture", "charge", "dépense", "depense", "finance", "trésorerie", "tresorerie"))
        return semantic_ok, "expected business semantics"

    if case.fact_check in {"lot_ref", "lot_followup", "lots_compare", "parcel_semantic", "stock_product", "stock_semantic", "inputs_semantic", "process_compare", "process_semantic", "material_semantic", "loss_semantic"}:
        semantic_ok = any(token in lower_blob for token in ("lot", "stock", "perte", "collect", "parcelle", "séch", "sech", "tri", "bilan", "mati"))
        return semantic_ok, f"expected semantic signal for {case.fact_check}"

    return True, "not_checked"


def _evaluate_case(
    case: ConversationCase,
    truth: dict[str, Any],
    payload: dict[str, Any],
    status_code: int,
    state: ConversationState,
) -> dict[str, Any]:
    issues: list[str] = []
    checks_total = 0
    checks_passed = 0

    if status_code != 200:
        return {
            "case_id": case.case_id,
            "module": case.module,
            "question": case.question,
            "status": "FAIL",
            "score": 0.0,
            "route": "",
            "source_types": [],
            "block_types": [],
            "issues": [f"HTTP_{status_code}"],
            "detected_entities": {},
            "answer_preview": str(payload)[:240],
        }

    route = _normalize_route(payload.get("route"))
    source_types = _source_types(payload)
    block_types = _block_types(payload)
    entities = _extract_entities(payload)
    answer = str(payload.get("answer") or "")

    checks_total += 1
    if route in {item.upper() for item in case.expected_routes}:
        checks_passed += 1
    else:
        issues.append(f"route_mismatch:{route} not in {sorted(case.expected_routes)}")

    checks_total += 1
    if case.required_sources.issubset(source_types):
        checks_passed += 1
    else:
        issues.append(f"missing_sources:{sorted(case.required_sources - source_types)}")

    checks_total += 1
    if {"summary", "sources"}.issubset(block_types):
        checks_passed += 1
    else:
        issues.append(f"missing_blocks:{sorted({'summary', 'sources'} - block_types)}")

    checks_total += 1
    fact_ok, fact_hint = _fact_check(case, truth, payload)
    if fact_ok:
        checks_passed += 1
    else:
        issues.append(f"fact_mismatch:{fact_hint}")

    checks_total += 1
    if _is_french_like(answer):
        checks_passed += 1
    else:
        issues.append("french_quality_weak")

    checks_total += 1
    if not (FALLBACK_MARKER in answer.lower() and source_types):
        checks_passed += 1
    else:
        issues.append("false_fallback_with_evidence")

    if case.expect_batch_reuse:
        checks_total += 1
        batch_ref = str(entities.get("batch_ref") or "").strip().upper()
        expected_ref = str(state.last_batch_ref or "").strip().upper()
        if batch_ref and expected_ref and batch_ref == expected_ref:
            checks_passed += 1
        else:
            issues.append(f"memory_batch_reuse_failed: expected={expected_ref or 'n/a'} got={batch_ref or 'none'}")

    if case.expect_context_reset:
        checks_total += 1
        if not str(entities.get("batch_ref") or "").strip():
            checks_passed += 1
        else:
            issues.append("context_reset_failed: batch_ref leaked")

    if case.expected_product:
        checks_total += 1
        if _expected_product_in_entities(entities, case.expected_product):
            checks_passed += 1
        else:
            issues.append(f"memory_product_missing:{case.expected_product}")

    if not issues:
        grade = "PASS"
    else:
        score_preview = checks_passed / checks_total if checks_total else 0.0
        if score_preview >= 0.55:
            grade = "PARTIAL"
        else:
            grade = "FAIL"

    # update conversation state from explicit or extracted batch refs
    question_batch_match = LOT_CODE_PATTERN.search(case.question or "")
    if question_batch_match:
        state.last_batch_ref = str(question_batch_match.group(0)).upper()
    elif str(entities.get("batch_ref") or "").strip():
        state.last_batch_ref = str(entities.get("batch_ref") or "").strip().upper()

    score = checks_passed / checks_total if checks_total else 0.0
    return {
        "case_id": case.case_id,
        "module": case.module,
        "question": case.question,
        "status": grade,
        "score": round(score, 4),
        "route": route,
        "source_types": sorted(source_types),
        "block_types": sorted(block_types),
        "issues": issues,
        "detected_entities": entities,
        "answer_preview": answer[:420],
    }


def _member_ranking_diagnosis(truth: dict[str, Any], ranking_case_result: dict[str, Any] | None) -> dict[str, Any]:
    top = (truth.get("top_members") or [None])[0]
    expected_name = str((top or {}).get("full_name") or "")
    expected_kg = float((top or {}).get("total_kg") or 0.0)
    legacy_name = "Awa Ndiaye"
    legacy_kg = 1715.0

    if not ranking_case_result:
        return {
            "verdict": "insufficient_data",
            "expected_top_member": expected_name,
            "expected_top_kg": expected_kg,
            "legacy_expectation": {"name": legacy_name, "kg": legacy_kg},
            "details": "ranking case result missing",
        }

    preview = str(ranking_case_result.get("answer_preview") or "").lower()
    chatbot_mentions_expected = expected_name.lower() in preview if expected_name else False
    chatbot_mentions_legacy = legacy_name.lower() in preview
    chatbot_mentions_expected_kg = _contains_close_number(preview, expected_kg, tolerance_ratio=0.12, tolerance_abs=2.0)

    if chatbot_mentions_expected and chatbot_mentions_expected_kg:
        if expected_name.lower() != legacy_name.lower() or abs(expected_kg - legacy_kg) > 1.0:
            verdict = "stale_audit_expectation"
            details = "chatbot aligns with current SQL truth using coop filters; legacy expectation appears stale"
        else:
            verdict = "no_mismatch_detected"
            details = "chatbot and legacy expectation align"
    else:
        if chatbot_mentions_legacy and expected_name and expected_name.lower() != legacy_name.lower():
            verdict = "chatbot_bug"
            details = "chatbot appears aligned with legacy output rather than current SQL truth"
        else:
            verdict = "chatbot_bug_or_parsing_gap"
            details = "chatbot response did not clearly match SQL truth for top member ranking"

    return {
        "verdict": verdict,
        "expected_top_member": expected_name,
        "expected_top_kg": expected_kg,
        "legacy_expectation": {"name": legacy_name, "kg": legacy_kg},
        "chatbot_mentions_expected": chatbot_mentions_expected,
        "chatbot_mentions_expected_kg": chatbot_mentions_expected_kg,
        "chatbot_mentions_legacy": chatbot_mentions_legacy,
        "details": details,
    }


def _build_summary(results: list[dict[str, Any]], ranking_diag: dict[str, Any]) -> dict[str, Any]:
    total = len(results)
    pass_count = sum(1 for row in results if row["status"] == "PASS")
    partial_count = sum(1 for row in results if row["status"] == "PARTIAL")
    fail_count = sum(1 for row in results if row["status"] == "FAIL")

    by_module: dict[str, dict[str, int]] = {}
    grouped: dict[str, list[dict[str, Any]]] = defaultdict(list)
    for row in results:
        grouped[row["module"]].append(row)
    for module, rows in grouped.items():
        by_module[module] = {
            "total": len(rows),
            "PASS": sum(1 for x in rows if x["status"] == "PASS"),
            "PARTIAL": sum(1 for x in rows if x["status"] == "PARTIAL"),
            "FAIL": sum(1 for x in rows if x["status"] == "FAIL"),
        }

    memory_followup_cases = [row for row in results if any(issue.startswith("memory_batch_reuse_failed") for issue in row.get("issues", [])) or row["case_id"] in {"lots-02", "lots-03", "ml-04", "reco-03"}]
    memory_followup_total = len([row for row in results if row["case_id"] in {"lots-02", "lots-03", "ml-04", "reco-03"}])
    memory_followup_pass = sum(1 for row in results if row["case_id"] in {"lots-02", "lots-03", "ml-04", "reco-03"} and not any(issue.startswith("memory_batch_reuse_failed") for issue in row.get("issues", [])))

    reset_total = len([row for row in results if row["case_id"] in {"members-01", "rag-01", "stocks-01"}])
    reset_pass = sum(1 for row in results if row["case_id"] in {"members-01", "rag-01", "stocks-01"} and "context_reset_failed: batch_ref leaked" not in row.get("issues", []))

    false_fallback_count = sum(1 for row in results if "false_fallback_with_evidence" in row.get("issues", []))
    french_quality_fail_count = sum(1 for row in results if "french_quality_weak" in row.get("issues", []))

    issue_counter = Counter()
    for row in results:
        for issue in row.get("issues", []):
            issue_counter[issue.split(":", 1)[0]] += 1

    worst = sorted(results, key=lambda row: (row["status"] != "FAIL", row["score"]))[:10]

    rating = round(((pass_count * 1.0 + partial_count * 0.6) / total) * 100, 1) if total else 0.0

    top_fix_map = {
        "route_mismatch": "Renforcer les règles sémantiques de routage multi-intent et follow-up.",
        "missing_sources": "Garantir la collecte de sources attendues par route (SQL/RAG/ML/reco).",
        "missing_blocks": "Uniformiser la production de response_blocks summary/sources/table par agent.",
        "fact_mismatch": "Ajouter des vérifications de vérité SQL avant formulation finale (post-aggregation check).",
        "memory_batch_reuse_failed": "Améliorer la résolution référentielle batch pour les pronoms et follow-ups courts.",
        "context_reset_failed": "Renforcer les garde-fous de reset de contexte lors des changements de module.",
        "false_fallback_with_evidence": "Interdire tout fallback générique dès qu’une preuve exploitable existe.",
        "french_quality_weak": "Améliorer le style de génération FR sous contraintes multi-agent.",
    }
    top_issues = issue_counter.most_common(5)
    top_fixes = [top_fix_map.get(key, f"Traiter les occurrences de {key}.") for key, _ in top_issues]

    return {
        "total_cases": total,
        "pass_count": pass_count,
        "partial_count": partial_count,
        "fail_count": fail_count,
        "pass_rate": (pass_count / total) if total else 0.0,
        "overall_verdict": "PASS" if fail_count == 0 else "PARTIAL" if fail_count <= max(1, total // 10) else "FAIL",
        "metrics_by_module": by_module,
        "memory_metrics": {
            "followup_pass": memory_followup_pass,
            "followup_total": memory_followup_total,
            "context_reset_pass": reset_pass,
            "context_reset_total": reset_total,
            "context_leak_count": max(reset_total - reset_pass, 0),
        },
        "false_fallback_count": false_fallback_count,
        "french_quality_fail_count": french_quality_fail_count,
        "issue_counts": dict(issue_counter),
        "worst_failures": [
            {
                "case_id": row["case_id"],
                "module": row["module"],
                "status": row["status"],
                "score": row["score"],
                "issues": row["issues"],
            }
            for row in worst
            if row["status"] in {"FAIL", "PARTIAL"}
        ][:10],
        "suspected_root_causes": [item[0] for item in top_issues],
        "chatbot_rating_over_100": rating,
        "top_5_recommended_fixes": top_fixes[:5],
        "member_ranking_diagnosis": ranking_diag,
    }


def _build_markdown(payload: dict[str, Any]) -> str:
    summary = payload["summary"]
    lines = [
        "# Chatbot Conversation Flow Audit (Supabase Readonly)",
        "",
        f"- Timestamp UTC: {payload['timestamp_utc']}",
        f"- Conversation ID: {payload['conversation_id']}",
        f"- Total cases: {summary['total_cases']}",
        f"- PASS: {summary['pass_count']}",
        f"- PARTIAL: {summary['partial_count']}",
        f"- FAIL: {summary['fail_count']}",
        f"- PASS rate: {summary['pass_rate']:.1%}",
        f"- Overall verdict: {summary['overall_verdict']}",
        f"- Chatbot rating: {summary['chatbot_rating_over_100']}/100",
        "",
        "## Metrics by module",
        "| Module | Total | PASS | PARTIAL | FAIL |",
        "|---|---:|---:|---:|---:|",
    ]
    for module, stats in summary["metrics_by_module"].items():
        lines.append(f"| {module} | {stats['total']} | {stats['PASS']} | {stats['PARTIAL']} | {stats['FAIL']} |")

    mm = summary["memory_metrics"]
    lines.extend(
        [
            "",
            "## Memory/context metrics",
            f"- Follow-up reuse: {mm['followup_pass']} / {mm['followup_total']}",
            f"- Context reset: {mm['context_reset_pass']} / {mm['context_reset_total']}",
            f"- Context leak count: {mm['context_leak_count']}",
            f"- False fallback count: {summary['false_fallback_count']}",
            "",
            "## Member ranking diagnosis",
            f"- Verdict: {summary['member_ranking_diagnosis']['verdict']}",
            f"- SQL truth top member: {summary['member_ranking_diagnosis']['expected_top_member']} ({summary['member_ranking_diagnosis']['expected_top_kg']:.1f} kg)",
            f"- Legacy expectation: {summary['member_ranking_diagnosis']['legacy_expectation']['name']} ({summary['member_ranking_diagnosis']['legacy_expectation']['kg']:.1f} kg)",
            f"- Detail: {summary['member_ranking_diagnosis']['details']}",
            "",
            "## Worst failures",
        ]
    )
    if summary["worst_failures"]:
        for item in summary["worst_failures"]:
            lines.append(f"- {item['case_id']} [{item['status']}] {item['module']} -> {', '.join(item['issues'])}")
    else:
        lines.append("- None")

    lines.extend(["", "## Suspected root causes"])
    if summary["suspected_root_causes"]:
        for item in summary["suspected_root_causes"]:
            lines.append(f"- {item}")
    else:
        lines.append("- None")

    lines.extend(["", "## Top 5 recommended fixes"])
    for idx, item in enumerate(summary["top_5_recommended_fixes"], start=1):
        lines.append(f"{idx}. {item}")

    lines.extend(["", "## Case log"])
    for row in payload["cases"]:
        lines.append(f"- {row['case_id']} [{row['status']}] module={row['module']} route={row['route']} sources={','.join(row['source_types'])}")
        if row["issues"]:
            lines.append(f"  - issues: {', '.join(row['issues'])}")

    return "\n".join(lines)
